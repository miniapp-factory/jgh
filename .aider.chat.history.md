function shuffleArray<T>(array: T[]): T[] {
  const arr = [...array];
  for (let i = arr.length - 1; i > 0; i--) {

'use client';
import { useEffect, useState, useRef } from "react";

interface Obstacle {
  x: number;
}

export default function JumpDodgeGame() {
  const [score, setScore] = useState(0);
  const [isJumping, setIsJumping] = useState(false);
  const [gameOver, setGameOver] = useState(false);
  const [obstacles, setObstacles] = useState<Obstacle[]>([]);
  const obstacleTimer = useRef<NodeJS.Timeout | null>(null);
  const gameTimer = useRef<NodeJS.Timeout | null>(null);

  // Handle jump
  const handleJump = () => {
    if (gameOver) return;
    setIsJumping(true);
    setTimeout(() => setIsJumping(false), 200);
  };

  // Add new obstacle at random intervals
  useEffect(() => {
    if (gameOver) return;
    obstacleTimer.current = setInterval(() => {
      setObstacles((prev) => [...prev, { x: 100 }]);
    }, 2000);
    return () => {
      if (obstacleTimer.current) clearInterval(obstacleTimer.current);
    };
  }, [gameOver]);

  // Game loop
  useEffect(() => {
    if (gameOver) return;
    gameTimer.current = setInterval(() => {
      setScore((s) => s + 1);

      setObstacles((prev) =>
        prev
          .map((o) => ({ ...o, x: o.x - 5 }))
          .filter((o) => {
            if (o.x <= 0) {
              if (isJumping) {
                // Successfully jumped over
                return false;
              } else {
                // Hit obstacle
                setGameOver(true);
                return false;
              }
            }
            return true;
          })
      );
    }, 100);

    return () => {
      if (gameTimer.current) clearInterval(gameTimer.current);
    };
  }, [isJumping, gameOver]);

  return (
    <div
      className="relative w-[300px] h-[200px] bg-gray-200 border border-gray-400 rounded-md"
      onClick={handleJump}
    >
      <div className="absolute bottom-0 left-0 w-4 h-4 bg-blue-500 rounded-full" />
      {obstacles.map((o, idx) => (
        <div
          key={idx}
          className="absolute bottom-0 w-4 h-4 bg-red-500 rounded-full"

**Solution Explanation**

For every test case we are given

* `N` – number of students  
* `M` – number of pairs of friends  
* `K` – number of students that are already in the club

The `M` pairs are undirected edges of a graph whose vertices are the
students.  
Two students are in the same connected component of this graph
iff they can reach each other by a chain of friendships.

The club can invite at most `K` students.
If a student is invited, **all** students of his/her connected component
must be invited as well.
Therefore the club can invite a whole component only if its size is
`≤ K`.  
The task is to count how many connected components satisfy this
condition.



--------------------------------------------------------------------

#### Algorithm
For each test case

1. Build the adjacency list of the graph.
2. Initialise an array `visited[1…N]` with `false`.
3. For every vertex `v` from `1` to `N`
   * if `visited[v]` is `false`
     * run a DFS/BFS starting from `v`
     * count the number of vertices in this component (`cnt`)
     * if `cnt ≤ K` increment the answer.
4. Output the answer.

--------------------------------------------------------------------

#### Correctness Proof  

We prove that the algorithm outputs the required number of components.

---

##### Lemma 1  
During the traversal started from a vertex `v` the algorithm visits
exactly all vertices that belong to the connected component of `v`.

**Proof.**

The traversal follows only edges of the graph.
Because the graph is undirected, from any vertex in the component
there is a path to `v`, and the traversal will follow that path.
Conversely, the traversal never leaves the component because it never
uses an edge that connects to a vertex outside the component.
∎



##### Lemma 2  
For every connected component `C` of the graph the algorithm
computes `cnt = |C|` exactly once.

**Proof.**

Let `v` be the first vertex of `C` that is processed
(`visited[v]` is `false`).
By Lemma&nbsp;1 the traversal started from `v` visits all vertices of `C`,
so `cnt = |C|`.  
After the traversal all vertices of `C` are marked visited,
hence no later vertex of `C` will start another traversal.
Thus `cnt` is computed exactly once for `C`. ∎



##### Lemma 3  
The algorithm increments the answer exactly for those components
whose size is at most `K`.

**Proof.**

By Lemma&nbsp;2 the algorithm obtains the exact size `cnt` of each
component.  
It increments the answer iff `cnt ≤ K`.  
Therefore the answer is increased precisely for the components
with size at most `K`. ∎



##### Theorem  
For each test case the algorithm outputs the number of connected
components that can be invited to the club.

**Proof.**

A component can be invited iff its size is at most `K`
(because the club can invite at most `K` students and must invite the
whole component).  
By Lemma&nbsp;3 the algorithm counts exactly those components.
Hence the printed number equals the required number of inviteable
components. ∎



--------------------------------------------------------------------

#### Complexity Analysis

Let `N` be the number of students and `M` the number of friendship
pairs.

* Building the adjacency list: `O(N + M)`
* Each vertex and each edge is processed once during DFS/BFS:
  `O(N + M)`

Total time per test case: `O(N + M)`  
Memory usage: adjacency list `O(N + M)` plus the visited array
`O(N)`.

import sys
from collections import deque

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    t = int(next(it))
    out_lines = []

    for _ in range(t):
        n = int(next(it))
        m = int(next(it))
        k = int(next(it))

        adj = [[] for _ in range(n + 1)]
        for _ in range(m):
            u = int(next(it))
            v = int(next(it))
            adj[u].append(v)
            adj[v].append(u)

        visited = [False] * (n + 1)
        answer = 0

        for v in range(1, n + 1):
            if not visited[v]:
                # BFS to count component size
                q = deque([v])
                visited[v] = True
                cnt = 0
                while q:
                    cur = q.popleft()
                    cnt += 1
                    for nb in adj[cur]:
                        if not visited[nb]:
                            visited[nb] = True
                            q.append(nb)
                if cnt <= k:
                    answer += 1

        out_lines.append(str(answer))

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
